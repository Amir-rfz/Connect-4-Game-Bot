class Connect4UI:
    def __init__(self, width=COLUMN_COUNT*SQUARESIZE, height=(ROW_COUNT+1)*SQUARESIZE):
        pygame.init()
        self.width = width
        self.height = height
        self.size = (self.width, self.height)
        self.screen = pygame.display.set_mode(self.size)
        self.font = pygame.font.SysFont("monospace", 75)

    def draw_board(self, board):
        for c in range(COLUMN_COUNT):
            for r in range(ROW_COUNT):
                pygame.draw.rect(self.screen, BLUE, (c * SQUARESIZE, r * SQUARESIZE + SQUARESIZE, SQUARESIZE, SQUARESIZE))
                pygame.draw.circle(self.screen, BLACK, (int(c * SQUARESIZE + SQUARESIZE / 2), int(r * SQUARESIZE + SQUARESIZE + SQUARESIZE / 2)), RADIUS)

        for c in range(COLUMN_COUNT):
            for r in range(ROW_COUNT):
                if board[r][c] == PLAYER_PIECE:
                    pygame.draw.circle(self.screen, RED, (int(c * SQUARESIZE + SQUARESIZE / 2), self.height - int(r * SQUARESIZE + SQUARESIZE / 2)), RADIUS)
                elif board[r][c] == CPU_PIECE:
                    pygame.draw.circle(self.screen, YELLOW, (int(c * SQUARESIZE + SQUARESIZE / 2), self.height - int(r * SQUARESIZE + SQUARESIZE / 2)), RADIUS)

        pygame.display.update()
        sleep(0.2)
        

    def display_winner(self, winner):
        if winner == PLAYER:
            label = self.font.render("Player wins!!", 1, RED)
        elif winner == CPU:
            label = self.font.render("Computer wins!!", 1, YELLOW)
        else:
            label = self.font.render("It's a draw!!", 1, BLUE)
        self.screen.blit(label, (40, 10))
        pygame.display.update()
        sleep(5)

class Connect4Game:
    def __init__(self, ui, minimax_depth, prune):
        self.board = np.zeros((ROW_COUNT, COLUMN_COUNT))
        self.ui = Connect4UI() if ui else None
        self.minimax_depth = minimax_depth
        self.prune = prune
        self.current_turn = random.choice([1, -1])
        self.nodes_seen_count = 0
    
    def drop_piece(self, board, row, col, piece):
        board[row][col] = piece

    def get_next_open_row(self, board,col):
        for r in range(ROW_COUNT):
            if board[r][col] == 0:
                return r
            
    def print_board(self, board):
        print(np.flip(board, 0))

    def winning_move(self, board, piece):
        for c in range(COLUMN_COUNT - 3):
            for r in range(ROW_COUNT):
                if all(board[r][c+i] == piece for i in range(WINDOW_LENGTH)):
                    return True

        for c in range(COLUMN_COUNT):
            for r in range(ROW_COUNT - 3):
                if all(board[r+i][c] == piece for i in range(WINDOW_LENGTH)):
                    return True

        for c in range(COLUMN_COUNT - 3):
            for r in range(ROW_COUNT - 3):
                if all(board[r+i][c+i] == piece for i in range(WINDOW_LENGTH)):
                    return True

        for c in range(COLUMN_COUNT - 3):
            for r in range(3, ROW_COUNT):
                if all(board[r-i][c+i] == piece for i in range(WINDOW_LENGTH)):
                    return True

        return False
    
    def evaluate_window(self, window, piece):
        score = 0
        opp_piece = PLAYER_PIECE
        if piece == PLAYER_PIECE:
            opp_piece = CPU_PIECE
        
        if window.count(piece) == 4:
            score += 100
        elif window.count(piece) == 3 and window.count(EMPTY) == 1:
            score += 5
        elif window.count(piece) == 2 and window.count(EMPTY) == 2:
            score += 2
        if window.count(opp_piece) == 3 and window.count(EMPTY) == 1:
            score -= 4
        
        return score
    
    def score_position(self, board, piece):
        score = 0  
        center_array = [int(i) for i in list(board[:, COLUMN_COUNT//2])]
        center_count = center_array.count(piece)
        score += center_count * 3
    
        for r in range(ROW_COUNT):
            row_array = [int(i) for i in list(board[r,:])]
            for c in range(COLUMN_COUNT-3):
                window = row_array[c:c+WINDOW_LENGTH]
                score += self.evaluate_window(window, piece)

        for c in range(COLUMN_COUNT):
            col_array = [int(i) for i in list(board[:,c])]
            for r in range(ROW_COUNT-3):
                window = col_array[r:r+WINDOW_LENGTH]
                score += self.evaluate_window(window, piece)

        for r in range(ROW_COUNT-3):
            for c in range(COLUMN_COUNT-3):
                window = [board[r+i][c+i] for i in range(WINDOW_LENGTH)]
                score += self.evaluate_window(window, piece)
                
        for r in range(ROW_COUNT-3):
            for c in range(COLUMN_COUNT-3):
                window = [board[r+3-i][c+i] for i in range(WINDOW_LENGTH)]
                score += self.evaluate_window(window, piece)
    
        return score
    
    def is_terminal_node(self, board):
        return self.winning_move(board, PLAYER_PIECE) or self.winning_move(board, CPU_PIECE) or len(self.get_valid_locations(board)) == 0
    
    def get_valid_locations(self, board):
        valid_locations = []
        for col in range(COLUMN_COUNT):
            if board[ROW_COUNT-1][col] == 0:
                valid_locations.append(col)
        return valid_locations
    
    def best_cpu_score(self, board, moves):
        move = None
        max_score = -math.inf
        for col in moves:
            row = self.get_next_open_row(board, col)
            b_copy = board.copy()
            self.drop_piece(b_copy, row, col, CPU_PIECE)
            score = self.heuristic(b_copy, CPU_PIECE)
            if score > max_score:
                max_score = score
                move = col
        return move
    
    def heuristic(self, board, piece):
        if(self.is_terminal_node(board)):
            if self.winning_move(board, piece):
                return WINING_SCORE
            elif self.winning_move(board, -piece):
                return -WINING_SCORE
            else:
                return 0
        else:
            return self.score_position(board, piece) - self.score_position(board, -piece)
        
    def minimax(self, board, depth, alpha, beta, player):
        self.nodes_seen_count += 1
        valid_locations = self.get_valid_locations(board)
        is_terminal = self.is_terminal_node(board)
        if depth == 0 or is_terminal:
            return None, self.heuristic(board, PLAYER_PIECE)

        if player == PLAYER_PIECE:
            max_score = -math.inf
            best_col = None

            for col in valid_locations:
                row = self.get_next_open_row(deepcopy(board), col)
                b_copy = deepcopy(board)
                self.drop_piece(b_copy, row, col, PLAYER_PIECE)
                _ , score = self.minimax(b_copy, depth - 1, alpha, beta, CPU_PIECE)
                if score > max_score:
                    max_score = score
                    best_col = col
                alpha = max(alpha, max_score)
                if self.prune and beta <= alpha:
                    break
            return best_col, max_score
        
        else:
            min_score = math.inf
            best_col = None

            for col in valid_locations:
                row = self.get_next_open_row(deepcopy(board), col)
                b_copy = deepcopy(board)
                self.drop_piece(b_copy, row, col, CPU_PIECE)
                _ , score = self.minimax(b_copy, depth - 1, alpha, beta, PLAYER_PIECE)
                if score < min_score:
                    min_score = score
                    best_col = col
                beta = min(beta, min_score)
                if beta <= alpha and self.prune:
                    break
            return best_col, min_score
        
    def get_cpu_move(self, board, randomness_percent=30):
        moves = self.get_valid_locations(board)
        if len(moves) == 0:
            return None
        random_move = random.choice(moves)
        score_move = self.best_cpu_score(board, moves)
        move = random.choice([score_move] * (100 - randomness_percent) + [random_move] * randomness_percent)
        return move, self.get_next_open_row(board, move)


    def get_human_move(self, board, depth = 1):
        col = self.minimax(board, depth, -math.inf, math.inf, self.current_turn)[0]
        return col, self.get_next_open_row(board, col)

    def play(self):
        self.nodes_seen_count = 0
        winner = None
        while not self.is_terminal_node(self.board):
            if self.ui:
                self.ui.draw_board(self.board)
            if self.current_turn == PLAYER:
                col, row = self.get_human_move(self.board, self.minimax_depth)
                if col is not None:
                    self.drop_piece(self.board, row, col, PLAYER_PIECE)
                    if self.winning_move(self.board, PLAYER_PIECE):
                        winner = PLAYER
                    self.current_turn = CPU
            else:
                col, row = self.get_cpu_move(self.board)
                if col is not None:
                    self.drop_piece(self.board, row, col, CPU_PIECE)
                    if self.winning_move(self.board, CPU_PIECE):
                        winner = CPU
                    self.current_turn = PLAYER
            
            if self.ui:
                self.ui.draw_board(self.board)
                if winner is not None:
                    self.ui.display_winner(winner)
                            
        if winner is None:
            winner = 0
        return winner
    
    def get_nodes_seen(self):
        return self.nodes_seen_count

Analysis of Connect 4 Game Performance by Depth¶

In this experiment, we evaluate the performance of the Connect 4 game AI across varying depths of the minimax algorithm without pruning. We aim to analyze three key metrics for each depth level:

  1. Average Runtime: The average time taken to compute each move.

  2. Probability of Winning: The likelihood of the player winning against the CPU.

  3. Average Nodes Seen: The average number of nodes explored by the minimax algorithm for each depth level.

import time
import matplotlib.pyplot as plt

def simulate_game(depth, pruning, runs):
    results = {-1: 0, 0: 0, 1: 0}
    total_time = 0
    total_nodes = 0

    for _ in range(runs):
        game = Connect4Game(False, depth, pruning)
        start_time = time.time()
        
        outcome = game.play()
        
        elapsed_time = time.time() - start_time
        nodes_seen = game.get_nodes_seen()
        
        results[outcome] += 1
        total_time += elapsed_time
        total_nodes += nodes_seen

    avg_time = total_time / runs
    win_prob = results[1] / runs
    avg_nodes = total_nodes / runs
    
    return avg_time, win_prob, avg_nodes

def run_analysis(depths, pruning, runs):
    times, win_probs, node_counts = [], [], []
    
    for depth in depths:
        avg_time, win_prob, avg_nodes = simulate_game(depth, pruning=pruning, runs=runs)
        
        times.append(avg_time)
        win_probs.append(win_prob)
        node_counts.append(avg_nodes)
        
        print(f"Depth {depth} -> Avg Time: {avg_time:.2f}s, Win Prob: {win_prob:.2f}, Avg Nodes: {avg_nodes}")

    # show plots
    plt.figure(figsize=(15, 5))
    
    plt.subplot(1, 3, 1)
    plt.plot(depths, times, marker='o')
    plt.title("Average Time vs Depth")
    plt.xlabel("Depth")
    plt.ylabel("Average Time (s)")

    plt.subplot(1, 3, 2)
    plt.plot(depths, win_probs, marker='o')
    plt.title("Winning Probability vs Depth")
    plt.xlabel("Depth")
    plt.ylabel("Probability of Winning")

    plt.subplot(1, 3, 3)
    plt.plot(depths, node_counts, marker='o')
    plt.title("Average Nodes Seen vs Depth")
    plt.xlabel("Depth")
    plt.ylabel("Average Nodes Seen")
    
    plt.tight_layout()
    plt.show()

Questions¶

Relationship Between Depth and Calculated Parameters in the Minimax Algorithm¶

  1. Depth and Calculated Parameters
    In the Minimax algorithm, the depth refers to how many layers of moves the algorithm explores before making a decision. As the depth increases:

    • Chance of Winning: A deeper search allows the algorithm to evaluate more potential future moves, which can improve the decision-making process and increase the chances of winning. However, this improvement is limited as deeper levels may not always provide significantly better decisions if the evaluation function is not accurate.

    • Time: The time complexity increases exponentially with depth. As the depth increases, the algorithm must evaluate more nodes, leading to longer computation times. This happens because the number of possible game states grows exponentially with each move (i.e., O(b^d), where b is the branching factor and d is the depth).

    • Nodes Seen: The number of nodes the algorithm needs to explore increases significantly with the depth. The search space grows exponentially, leading to more nodes being explored at higher depths. This is one of the main challenges of using Minimax in games with large search spaces.

    Therefore, increasing the depth enhances decision-making but at the cost of increased time and nodes explored, which may limit its practical application for real-time or resource-constrained environments.

Can We Choose the Order of Seeing the Children of Each Node to Maximize Pruning?¶

Yes, we can choose the order in which we explore the children of each node to maximize pruning, and this technique is known as Move Ordering.

  1. Pruning in Minimax Algorithm

    Pruning occurs when the algorithm skips evaluating certain branches of the game tree because it knows that these branches cannot affect the final decision. This is done through Alpha-Beta Pruning, where the algorithm keeps track of two values (alpha and beta) that represent the best possible scores for the maximizing and minimizing players, respectively.

  2. Method for Maximizing Pruning

    To maximize pruning, we need to explore the most promising moves first. If we order the children of each node such that the most likely moves to lead to a favorable outcome are explored first, the algorithm is more likely to prune unimportant branches earlier.

    • Maximizing Pruning: By evaluating the most promising moves first (those likely to lead to a high score for the maximizing player or low score for the minimizing player), the algorithm will find a good value for alpha and beta sooner, leading to more effective pruning of unimportant branches.

  3. How to Implement This?

    • Heuristic-based Ordering: One way to order the children is to use a heuristic evaluation function that estimates the potential value of each move before exploring the full game tree.

    • Transposition Tables: Another approach is using transposition tables to store previously computed values for certain game states, which helps in deciding which branches to explore.

    By ordering the children in this way, we can maximize pruning, making the algorithm more efficient and reducing the number of nodes explored.

Branching Factor and Its Changes with Game Development¶

  1. Branching Factor

    The branching factor refers to the number of possible moves (or children) that can be generated from a given node in the game tree. In the context of the Minimax algorithm, the branching factor determines the number of nodes that the algorithm needs to evaluate at each level of the search tree. A larger branching factor means more potential moves to explore, leading to higher computational complexity.

    For example, in a game like Connect 4, at each turn, the number of possible moves corresponds to the number of empty columns where a player can drop a piece. If the board is nearly full, the branching factor is smaller, but if many moves are available, the branching factor increases.

  2. Changes with Game Development

    As the game progresses and the player or the AI gains experience, the following changes may occur in the branching factor:

    • Early Game: In the beginning of the game, when the board is mostly empty, the branching factor is relatively high because many moves are possible. As players continue to make moves, the number of available moves gradually decreases.

    • Mid and Late Game: As more pieces are placed, the game space becomes more constrained, and the branching factor typically reduces. For the AI, focusing on key positions and reducing unnecessary exploration can help manage this.

    • AI-Driven Development: With further game development, especially incorporating AI with advanced techniques like Alpha-Beta Pruning and Heuristic Evaluation, the effective branching factor can be reduced, as many branches of the search tree may be pruned early, leading to more efficient decision-making.

    In summary, the branching factor will change dynamically depending on the game state, and strategies for pruning and move selection can help manage its impact on performance.

Why Does the Algorithm Become Faster Without Losing Accuracy When Pruning?¶

Pruning, specifically Alpha-Beta Pruning, makes the Minimax algorithm faster by eliminating branches in the game tree that cannot affect the final decision.

  1. How Pruning Works

    During Minimax search, Alpha-Beta Pruning keeps track of two values:

    • Alpha: the best score that the maximizing player can guarantee at that point.

    • Beta: the best score that the minimizing player can guarantee at that point.

    If, while exploring a node, it becomes clear that it will not affect the outcome (i.e., it falls outside the alpha-beta range), the algorithm prunes that branch, skipping further evaluation of that node.

  2. Improving Speed Without Losing Accuracy

    By pruning irrelevant branches, the algorithm can focus on critical moves while ignoring those that do not affect the final decision. This significantly reduces the number of nodes evaluated, making the algorithm faster without affecting the accuracy of the outcome, as it still explores all possible moves that could change the result.

In summary, pruning enhances the efficiency of the Minimax algorithm, allowing it to search deeper or run faster by ignoring branches that do not influence the final decision.

Why Use Minimax When the Opponent Acts by Chance? Is It the Optimal Method?¶

  1. Limitations of Minimax with Random Opponents

    Minimax is designed to anticipate an opponent's optimal moves by assuming they make decisions to maximize their own advantage. However, when the opponent acts randomly or unpredictably, Minimax may not be the most effective method. Since it assumes the opponent will make the best move, it may lead to suboptimal decision-making against a non-strategic, chance-driven opponent.

  2. Alternative Algorithm: Expectimax

    A better-suited algorithm for situations where the opponent’s moves are random is Expectimax:

    • How Expectimax Works: Instead of assuming an optimal opponent, Expectimax accounts for the probabilities of each possible move. It evaluates moves by calculating the expected utility, averaging the values based on the probability of each potential random move.

    • Advantages Over Minimax: Expectimax is more appropriate for games involving randomness, as it can handle both strategic moves and chance-based events, leading to more realistic and adaptive decisions.

In summary, Expectimax is often a better choice than Minimax in scenarios where an opponent behaves unpredictably or moves randomly, as it provides a probability-weighted evaluation that aligns better with the stochastic nature of the opponent's moves.
